67 struct nrlmsise_flags {
68  int switches[24];
69  double sw[24];
70  double swc[24];
71 };
72 /*
73  * Switches: to turn on and off particular variations use these switches.
74  * 0 is off, 1 is on, and 2 is main effects off but cross terms on.
75  *
76  * Standard values are 0 for switch 0 and 1 for switches 1 to 23. The
77  * array "switches" needs to be set accordingly by the calling program.
78  * The arrays sw and swc are set internally.
79  *
80  * switches[i]:
81  * i - explanation
82  * -----------------
83  * 0 - output in centimeters instead of meters
84  * 1 - F10.7 effect on mean
85  * 2 - time independent
86  * 3 - symmetrical annual
87  * 4 - symmetrical semiannual
88  * 5 - asymmetrical annual
89  * 6 - asymmetrical semiannual
90  * 7 - diurnal
91  * 8 - semidiurnal
92  * 9 - daily ap [when this is set to -1 (!) the pointer
93  * ap_a in struct nrlmsise_input must
94  * point to a struct ap_array]
95  * 10 - all UT/long effects
96  * 11 - longitudinal
97  * 12 - UT and mixed UT/long
98  * 13 - mixed AP/UT/LONG
99  * 14 - terdiurnal
100  * 15 - departures from diffusive equilibrium
101  * 16 - all TINF var
102  * 17 - all TLB var
103  * 18 - all TN1 var
104  * 19 - all S var
105  * 20 - all TN2 var
106  * 21 - all NLB var
107  * 22 - all TN3 var
108  * 23 - turbo scale height var
109  */
110 
111 struct ap_array {
112  double a[7];
113 };
114 /* Array containing the following magnetic values:
115  * 0 : daily AP
116  * 1 : 3 hr AP index for current time
117  * 2 : 3 hr AP index for 3 hrs before current time
118  * 3 : 3 hr AP index for 6 hrs before current time
119  * 4 : 3 hr AP index for 9 hrs before current time
120  * 5 : Average of eight 3 hr AP indicies from 12 to 33 hrs
121  * prior to current time
122  * 6 : Average of eight 3 hr AP indicies from 36 to 57 hrs
123  * prior to current time
124  */

127 struct nrlmsise_input {
128  int year; /* year, currently ignored */
129  int doy; /* day of year */
130  double sec; /* seconds in day (UT) */
131  double alt; /* altitude in kilometes */
132  double g_lat; /* geodetic latitude */
133  double g_long; /* geodetic longitude */
134  double lst; /* local apparent solar time (hours), see note below */
135  double f107A; /* 81 day average of F10.7 flux (centered on doy) */
136  double f107; /* daily F10.7 flux for previous day */
137  double ap; /* magnetic index(daily) */
138  struct ap_array *ap_a; /* see above */
139 };
140 /*
141  * NOTES ON INPUT VARIABLES:
142  * UT, Local Time, and Longitude are used independently in the
143  * model and are not of equal importance for every situation.
144  * For the most physically realistic calculation these three
145  * variables should be consistent (lst=sec/3600 + g_long/15).
146  * The Equation of Time departures from the above formula
147  * for apparent local time can be included if available but
148  * are of minor importance.
149  *
150  * f107 and f107A values used to generate the model correspond
151  * to the 10.7 cm radio flux at the actual distance of the Earth
152  * from the Sun rather than the radio flux at 1 AU. The following
153  * site provides both classes of values:
154  * ftp://ftp.ngdc.noaa.gov/STP/SOLAR_DATA/SOLAR_RADIO/FLUX/
155  *
156  * f107, f107A, and ap effects are neither large nor well
157  * established below 80 km and these parameters should be set to
158  * 150., 150., and 4. respectively.
159  */

167 struct nrlmsise_output {
168  double d[9]; /* densities */
169  double t[2]; /* temperatures */
170 };
171 /*
172  * OUTPUT VARIABLES:
173  * d[0] - HE NUMBER DENSITY(CM-3)
174  * d[1] - O NUMBER DENSITY(CM-3)
175  * d[2] - N2 NUMBER DENSITY(CM-3)
176  * d[3] - O2 NUMBER DENSITY(CM-3)
177  * d[4] - AR NUMBER DENSITY(CM-3)
178  * d[5] - TOTAL MASS DENSITY(GM/CM3) [includes d[8] in td7d]
179  * d[6] - H NUMBER DENSITY(CM-3)
180  * d[7] - N NUMBER DENSITY(CM-3)
181  * d[8] - Anomalous oxygen NUMBER DENSITY(CM-3)
182  * t[0] - EXOSPHERIC TEMPERATURE
183  * t[1] - TEMPERATURE AT ALT
184  *
185  *
186  * O, H, and N are set to zero below 72.5 km
187  *
188  * t[0], Exospheric temperature, is set to global average for
189  * altitudes below 120 km. The 120 km gradient is left at global
190  * average value for altitudes below 72 km.
191  *
192  * d[5], TOTAL MASS DENSITY, is NOT the same for subroutines GTD7
193  * and GTD7D
194  *
195  * SUBROUTINE GTD7 -- d[5] is the sum of the mass densities of the
196  * species labeled by indices 0-4 and 6-7 in output variable d.
197  * This includes He, O, N2, O2, Ar, H, and N but does NOT include
198  * anomalous oxygen (species index 8).
199  *
200  * SUBROUTINE GTD7D -- d[5] is the "effective total mass density
201  * for drag" and is the sum of the mass densities of all species
202  * in this model, INCLUDING anomalous oxygen.
203  */
